342 template <typename val_type>
343 val_type security_policy::validate(const std::string& key, const val_type& val) const
344 {
345  auto found = int_ranges_.find(key);
346  if (found == int_ranges_.end())
347  throw exception("The key " + key + " could not be found in the security policy, so its value cannot be tested");
348  if (!found->second->in_range(val))
349  {
350  std::ostringstream stream;
351  stream << "The value," << val << ", of key, " << key << ", lies outside the range," <<
352  found->second->to_text() << ", permitted by the security policy";
353  throw exception(stream.str());
354  }
355  return val;
356 }
