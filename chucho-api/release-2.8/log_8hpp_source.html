34 #define CHUCHO_LOG(lvl, lg, fl, ln, fnc, msg) \
35  do \
36  { \
37  if ((lg)->permits(lvl)) \
38  { \
39  std::ostringstream __chucho_internal_stream; \
40  __chucho_internal_stream << msg; \
41  (lg)->write(::chucho::event((lg), (lvl), __chucho_internal_stream.str(), (fl), (ln), (fnc))); \
42  } \
43  } while (false)

75 #define CHUCHO_EVERY_N_INTERNAL(n, body) \
76  do \
77  { \
78  static std::size_t __chucho_every_n_counter = 0; \
79  if (++__chucho_every_n_counter > (n)) \
80  __chucho_every_n_counter -= (n); \
